if(BUILD_SHARED_LIBS)
    target_link_libraries(${TEST_BINARY_NAME} ${BCOS_C_SDK_TARGET} Boost::unit_test_framework)
else ()
    target_link_libraries(${TEST_BINARY_NAME} ${BCOS_C_SDK_STATIC_TARGET} Boost::unit_test_framework)
endif ()
